    with open(path, 'r') as text:
        data = text.readlines()
    if linenumber <= len(data):
        l = data[linenumber-1].strip().split(',')
        # Here i use the map function:
        var, vek, strg = map(eval, [l[0], ','.join(l[1:-1]).strip(), l[-1]])
